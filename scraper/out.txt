function expandtaplayer() {
	var currentexp = $.cookie('exp_default');
	if(currentexp == 'expanded') {
		expanded = true;
	}
	if (typeof expanded === 'undefined') {
		$( "#side_ad" ).addClass( "side_ad_drop" );
		$( ".inner" ).addClass( "ta_expansion" );
		expanded = true;
		$.cookie('exp_default', 'expanded', { expires: 60, path: '/' });
	} else {
		$( "#side_ad" ).removeClass( "side_ad_drop" );
		$( ".inner" ).removeClass( "ta_expansion" );
		delete expanded;
		$.cookie('exp_default', '', { expires: 1, path: '/' });
	}
}
